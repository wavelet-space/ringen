Example 1: single producer, single consumerÔÉÅ

 1#include <iostream>
 2#include <ringen/ring_buffer.hpp>
 3#include <ringen/sequence_barrier.hpp>
 4#include <ringen/single_threaded_claim_strategy.hpp>
 5#include <ringen/spin_wait_strategy.hpp>
 6#include <thread>
 7
 8using namespace ringen;
 9
10struct event {
11  uint32_t data;
12};
13
14int main() {
15  constexpr size_t buffer_size = 1024 * 2; // Must be power-of-two!
16
17  ring_buffer<event> buffer(buffer_size);
18
19  spin_wait_strategy wait_strategy;
20  sequence_barrier<spin_wait_strategy> barrier(wait_strategy);
21
22  single_threaded_claim_strategy<spin_wait_strategy> claim_strategy(
23      buffer_size, wait_strategy);
24
25  claim_strategy.add_claim_barrier(barrier);
26
27  // The consumer thread lambda function.
28  std::thread consumer([&]() {
29    uint64_t sum = 0;
30    sequence_t next_to_read = 0;
31    bool done = false;
32
33    while (!done) {
34      // Wait until more items are available.
35      sequence_t available = claim_strategy.wait_until_published(next_to_read);
36
37      // Process all available items in a batch.
38      do {
39        auto &event = buffer[next_to_read];
40        sum += event.data;
41        if (event.data == 0) {
42          done = true;
43        }
44      } while (next_to_read++ != available);
45
46      // Notify producer we've finished consuming items.
47      barrier.publish(available);
48      std::cerr << "consumer: sum = " << sum
49                << "\r"; //  TODO Move print outside?
50    }
51  });
52
53  // The producer thread lambda function.
54  std::thread producer([&]() {
55    for (uint32_t i = 1; i <= 1'000'000; ++i) {
56      // Claim a slot in the ring buffer, waits if buffer is full.
57      sequence_t sequence = claim_strategy.claim_one();
58      // Write to the slot in the ring buffer.
59      buffer[sequence].data = i;
60      // Publish the event to the consumer.
61      claim_strategy.publish(sequence);
62      // std::cout << "produced: data = " <<  i << std::endl;
63    }
64    // Publish the terminating event.
65    sequence_t sequence = claim_strategy.claim_one();
66    buffer[sequence].data = 0;
67    claim_strategy.publish(sequence);
68  });
69
70  consumer.join();
71  producer.join();
72
73  return 0;
74}
